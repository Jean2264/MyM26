       public List<string> CamposDuplicados()
        {
            List<string> duplicados = new List<string>();
         


            string consulta = @"
        SELECT
         
            CASE WHEN EXISTS (SELECT 1 FROM Usuario WHERE Usuario = @Usuario ) THEN 'Sí' ELSE 'No' END AS Usuario_Duplicado;";

            using (SqlConnection conn = new SqlConnection(Decla.cnn.ConnectionString))
            using (SqlCommand cmd = new SqlCommand(consulta, conn))
            {
                cmd.Parameters.AddWithValue("@DNI", txt_dni.Text.Trim());
                cmd.Parameters.AddWithValue("@Usuario", txt_nombre.Text.Trim());


                try
                {
                    conn.Open();
                    SqlDataReader reader = cmd.ExecuteReader();
                    if (reader.Read())
                    {
                       
                        if (reader["Usuario_Duplicado"].ToString() == "Sí") duplicados.Add("Nombre de usuario");

                    }
                }
                catch (Exception ex)
                {
                    MessageBox.Show("Error al verificar duplicados:\n" + ex.Message, "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
                }
            }

            return duplicados;
        }

        private void IdMovimientos()
        {
            string consulta = "select MAX(IdHistorico) as UltimoId FROM HMovimiento";

            using (SqlConnection conn = new SqlConnection(Decla.cnn.ConnectionString))
            {
                SqlCommand cmd = new SqlCommand(consulta, conn);
                try
                {
                    conn.Open();
                    object obj = cmd.ExecuteScalar();
                    if (obj != DBNull.Value)
                    {
                        UltimoIdHisto = Convert.ToInt32(obj);
                    }
                    else
                    {
                        UltimoIdHisto = 0;
                    }
                }
                catch (Exception ex)
                {
                    MessageBox.Show("Error al traer el ultimo Id: " + ex.Message, "Aviso", MessageBoxButtons.OK, MessageBoxIcon.Error);
                }

            }
        }


         private void AltaHistorico()
        {
            NuevoIdHisto = 1 + UltimoIdHisto;
            CodHistorico = "MOV" + NuevoIdHisto;
            string consulta = "Insert into HMovimiento(IdHistorico, CodHistorico, DNI, TipoMovimiento, FechaHora, DetalleMovimiento) " +
                                               "VALUES(@IdHistorico, @CodHistorico, @DNI, @TipoMovimiento, GETDATE(), @DetalleMovimiento)";

            SqlCommand comd = new SqlCommand(consulta, Decla.cnn);

            comd.Parameters.AddWithValue("@IdHistorico", NuevoIdHisto);
            comd.Parameters.AddWithValue("@CodHistorico", CodHistorico);
            comd.Parameters.AddWithValue("@DNI", UsuarioActivo.DNI);
            comd.Parameters.AddWithValue("@TipoMovimiento", TipoMovimiento);
            comd.Parameters.AddWithValue("@DetalleMovimiento", DetalleMovimiento);
            try
            {
                Decla.cnn.Open();
                comd.ExecuteNonQuery();

            }
            catch (Exception ex)
            {
                MessageBox.Show("Error al cargar el movimiento: " + ex.Message);
            }
            finally
            {
                if (Decla.cnn.State == ConnectionState.Open)
                    Decla.cnn.Close();
            }
        }
