        private bool ValidarUsuario(string correo)
        {
            using (SqlConnection conn = new SqlConnection(Decla.cnn.ConnectionString))
            {
                conn.Open();
                string consulta = @"
                                SELECT COUNT(*) 
                                FROM Usuario s 
                                INNER JOIN TipoUsuario t ON s.CodTipoUsuario = t.CodTipoUsuario 
                                WHERE s.Mail = @Mail 
                                AND t.Tipo = 'Administrador' 
                                AND s.Estado = 1";

                SqlCommand cmd = new SqlCommand(consulta, conn);
                cmd.Parameters.AddWithValue("@Mail", correo);
                int count =Convert.ToInt32( cmd.ExecuteScalar());
                return count > 0;
            }
        }

        private string GenerarToken(int longitud = 8)

            const string chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
            var random = new Random();
            return new string(Enumerable.Repeat(chars, longitud).Select(s => s[random.Next(s.Length)]).ToArray());

        private bool ValidarToken(string correo, string tokenIngresado)
        {
            using (SqlConnection conn = new SqlConnection(Decla.cnn.ConnectionString))
            {
                conn.Open();
                string query = @"
                            SELECT s.TokenRecuperacion, s.TokenExpira, s.Usuario 
                            FROM Usuario s
                            INNER JOIN TipoUsuario t ON s.CodTipoUsuario = t.CodTipoUsuario
                            WHERE s.Mail = @correo
                            AND t.Tipo = 'Administrador'";
                SqlCommand cmd = new SqlCommand(query, conn);
                cmd.Parameters.AddWithValue("@correo", correo);

                SqlDataReader reader = cmd.ExecuteReader();
                if (reader.Read())
                {
                    string tokenDb = reader["TokenRecuperacion"].ToString();
                    DateTime expira = Convert.ToDateTime(reader["TokenExpira"]);
                    user = reader["Usuario"].ToString();

                    if (tokenDb == tokenIngresado && DateTime.Now <= expira)
                    {
                        return true;
                    }
                }
            }
            return false;
        }

        private void btn_validar_Click(object sender, EventArgs e)
        {
          
            errorProvider1.SetError(txt_pass, "");
            errorProvider1.SetError(txt_repet, "");

           
            if (string.IsNullOrWhiteSpace(txt_pass.Text))
            {
                errorProvider1.SetError(txt_pass, "La contraseña no puede estar vacía.");
                return;
            }
            if (txt_pass.Text.Length < 8 || txt_pass.Text.Length > 20)
            {
                errorProvider1.SetError(txt_pass, "La contraseña debe tener entre 8 y 20 caracteres.");
                return;
            }
            if (txt_pass.Text != txt_repet.Text)
            {
                errorProvider1.SetError(txt_repet, "Las contraseñas no coinciden.");
                return;
            }

           
            string consulta = @"UPDATE Usuario SET Contrasenia=@pass WHERE Usuario=@user AND Estado=1";

            try
            {
                using (SqlConnection cnn = new SqlConnection(Decla.cnn.ConnectionString))
                using (SqlCommand cmd = new SqlCommand(consulta, cnn))
                {
                    cmd.Parameters.Add("@pass", SqlDbType.VarChar, 20).Value = txt_repet.Text.ToLower();
                    cmd.Parameters.Add("@user", SqlDbType.VarChar, 50).Value = usuario;

                    cnn.Open();
                    cmd.ExecuteNonQuery();
                }

                MessageBox.Show("Contraseña actualizada con éxito ", "Éxito", MessageBoxButtons.OK, MessageBoxIcon.Information);

                Login lg = new Login();
                lg.StartPosition = FormStartPosition.CenterScreen;
                lg.Show();
                this.Close();
            }
            catch (Exception ex)
            {
                MessageBox.Show("Error al actualizar la contraseña: " + ex.Message, "Aviso", MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
        }
